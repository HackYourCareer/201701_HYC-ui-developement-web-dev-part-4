Sass - CSS na sterydach

Sass - CSS na sterydach

                
    // Awful CSS

    .italy .tower-of-pisa {
      font-style: italic;
    }
                
                
            
                
  // "Normal" CSS                   // Nesting 

  .italy .tower-of-pisa {           .italy {
    font-style: italic;               .tower-of-pisa {
  }                                     font-style: italic;
                                      }
                                    }
                
                
            
                
    // Variables

    $tower-color: white;

    .italy {
      .tower-of-pisa {
        font-style: italic;
        color: $tower-color;
      }
    }
                
                
            
                
    // Mixins

    $tower-color: white;

    @mixin leaning-tower() {
      font-style: italic;
      color: $tower-color;
    }

    .italy {
      .tower-of-pisa {
        @include leaning-tower();
      }
    }
