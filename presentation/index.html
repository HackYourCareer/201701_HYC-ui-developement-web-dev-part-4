      }
                
                
            
                
    .italy-monuments {
      .tower-of-pisa {
        font-style: italic;
      }
    }
                
                
            
                
    $tower-color: white;

    .italy-monuments {
      .tower-of-pisa {
        font-style: italic;
        color: $tower-color;
      }
    }
                
                
            
                
    $tower-color: white;

    @mixin leaning-tower() {
      font-style: italic;
      color: $tower-color;
    }

    .italy-monuments {
      .tower-of-pisa {
        @include leaning-tower();
      }
    }
